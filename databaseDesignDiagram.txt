// Doctor Prescription Digitization System (Scalable Design)
// DBML for dbdiagram.io

Table doctors {
  id integer [primary key, increment]
  name varchar [not null]
  email varchar [unique, not null]
  password_hash varchar [not null]
  specialization varchar [not null]
  bmdc_reg_no varchar [not null]
  education varchar [not null]
  doctor_chamber varchar [not null]
  visit_fee decimal(10,2) [not null]
  created_at timestamp [default: `now()`]
}

Table patients {
  id integer [primary key, increment]
  name varchar [not null]
  age integer [not null]
  gender enum('male','female','other') [not null]
  phone varchar [unique, not null]
  email varchar [null]
  weight decimal(5,2) [null]
  created_at timestamp [default: `now()`]
}

Table prescriptions {
  id integer [primary key, increment]
  doctor_id integer [not null]
  patient_id integer [not null]
  chief_complaints text [null]
  diagnosis text [null]
  investigation text [null]
  advice text [null]
  pdf_url varchar [null]
  created_at timestamp [default: `now()`]
}

Table medicines {
  id integer [primary key, increment]
  prescription_id integer [not null]
  medicine_name varchar [not null]
  dosage varchar [not null]
  timing varchar [not null]
  duration varchar [not null]
  notes text [null]
}

Table prescription_emails {
  id integer [primary key, increment]
  prescription_id integer [not null]
  recipient_email varchar [not null]
  sent_at timestamp [default: `now()`]
  status varchar [not null]
}

Table labs {
  id integer [primary key, increment]
  name varchar [not null]
  email varchar [unique, not null]
  password_hash varchar [not null]
  phone varchar [not null]
  address varchar [not null]
  license_no varchar [not null]
  created_at timestamp [default: `now()`]
}

Table medical_reports {
  id integer [primary key, increment]
  lab_id integer [not null]
  patient_id integer [not null]
  report_type enum('xray','mri','ct_scan','blood_test','ultrasound','other') [not null]
  title varchar [not null]
  description text [null]
  file_url varchar [not null]
  file_type enum('image','pdf') [not null]
  original_filename varchar [not null]
  report_date date [not null]
  created_at timestamp [default: `now()`]
}

// Relationships
Ref doctor_prescriptions: prescriptions.doctor_id > doctors.id
Ref patient_prescriptions: prescriptions.patient_id > patients.id
Ref prescription_medicines: medicines.prescription_id > prescriptions.id
Ref prescription_email_log: prescription_emails.prescription_id > prescriptions.id
Ref lab_reports: medical_reports.lab_id > labs.id
Ref patient_reports: medical_reports.patient_id > patients.id
